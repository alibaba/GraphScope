681

686  std::numeric_limits<INDEX_T>::max());
687  lf.indices_size_ = input.indices_.size();
688

690  input.hash_policy_.get_mod_function_index());
691  lf.num_slots_minus_one_ = input.num_slots_minus_one_;
692 
693  for (auto oid : input.keys_) {
694  size_t index = input.hash_policy_.index_for_hash(
695  input.hasher_(oid), input.num_slots_minus_one_);
696  for (int8_t distance = 0; input.distances_[index] >= distance;
697  ++distance, ++index) {

700  lf.indices_[index] = ret;
701  break;
702  }

704  }
