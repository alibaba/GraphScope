99  if (!indexer.add(casted_array->Value(i), vid)) {
100  LOG(FATAL) << "Duplicate vertex id: " << casted_array->Value(i)
101  << "..";
102  }
103  vids.emplace_back(vid);
104  }
105  } else {

107  auto casted_array = std::static_pointer_cast<arrow::StringArray>(col);
108  for (size_t i = 0; i < row_num; ++i) {
109  auto str = casted_array->GetView(i);

111  if (!indexer.add(str_view, vid)) {
112  LOG(FATAL) << "Duplicate vertex id: " << str_view << "..";
113  }
114  vids.emplace_back(vid);
115  }
116  } else if (col->type()->Equals(arrow::large_utf8())) {
117  auto casted_array =
118  std::static_pointer_cast<arrow::LargeStringArray>(col);
119  for (size_t i = 0; i < row_num; ++i) {
