144  if (oe->get_neighbor() == dst) {
145  oe->set_data(prop, ts);
146  src_flag = true;
147  break;
148  }
149  oe->next();
150  }
151  auto ie = in_csr_->edge_iter_mut(dst);
152  while (ie != nullptr && ie->is_valid()) {
153  if (ie->get_neighbor() == src) {
154  dst_flag = true;
155  ie->set_data(prop, ts);
156  break;
157  }
158  ie->next();
159  }
160  if (!(src_flag || dst_flag)) {
161  in_csr_->put_edge(dst, src, prop, ts, alloc);
