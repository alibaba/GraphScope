163  void open_with_hugepages(const std::string& filename, size_t capacity = 0) {
164  reset();
165  hugepage_prefered_ = true;
166  if (!filename.empty() && std::filesystem::exists(filename)) {
167  size_t file_size = std::filesystem::file_size(filename);
168  size_ = file_size / sizeof(T);
169  if (size_ != 0) {
170  capacity = std::max(capacity, size_);
171  mmap_size_ = hugepage_round_up(capacity * sizeof(T));
172  data_ = static_cast<T*>(allocate_hugepages(mmap_size_));
173  if (data_ != MAP_FAILED) {
174  FILE* fin = fopen(filename.c_str(), "rb");
175  CHECK_EQ(fread(data_, sizeof(T), size_, fin), size_);
176  fclose(fin);
177  } else {
178  LOG(ERROR) << "allocating hugepage failed, " << strerror(errno)
179  << ", try with normal pages";
180  open(filename, false);
181  }
