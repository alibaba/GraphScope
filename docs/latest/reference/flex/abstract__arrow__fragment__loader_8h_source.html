264  } else if (cur_indexer.get_type() == PropertyType::kUInt32) {
265  CHECK(cur_col->type()->Equals(arrow::uint32()));
266  } else if (cur_indexer.get_type() == PropertyType::kUInt64) {
267  CHECK(cur_col->type()->Equals(arrow::uint64()));
268  }
269  };

271  indexer_check_lambda(src_indexer, src_col);
272  indexer_check_lambda(dst_indexer, dst_col);
273  auto old_size = parsed_edges.size();
274  parsed_edges.resize(old_size + src_col->length());
275  VLOG(10) << "resize parsed_edges from" << old_size << " to "
276  << parsed_edges.size();

278  // if EDATA_T is grape::EmptyType, no need to read columns
279  auto edata_col_thread = std::thread([&]() {
280  if constexpr (std::is_same<EDATA_T, RecordView>::value) {
281  size_t cur_ind = old_size;
282  for (auto j = 0; j < src_col->length(); ++j) {
283  std::get<2>(parsed_edges[cur_ind++]) = offset++;

294  }
295 
296  using arrow_array_type =
297  typename gs::TypeConverter<EDATA_T>::ArrowArrayType;
298  // cast chunk to EDATA_T array
299  auto data = std::static_pointer_cast<arrow_array_type>(edata_col);
300  for (auto j = 0; j < edata_col->length(); ++j) {
301  if constexpr (std::is_same<arrow_array_type,
302  arrow::StringArray>::value ||
303  std::is_same<arrow_array_type,
304  arrow::LargeStringArray>::value) {
305  auto str = data->GetView(j);
306  std::string_view str_view(str.data(), str.size());
307  std::get<2>(parsed_edges[cur_ind++]) = str_view;
308  } else {
309  std::get<2>(parsed_edges[cur_ind++]) = data->Value(j);

314  });

316  auto src_col_thread = std::thread([&]() {
317  _append<SRC_PK_T, EDATA_T>(false, cur_ind, src_col, src_indexer,
318  parsed_edges, oe_degree);
319  });
320  auto dst_col_thread = std::thread([&]() {
321  _append<DST_PK_T, EDATA_T>(true, cur_ind, dst_col, dst_indexer,
322  parsed_edges, ie_degree);
323  });
