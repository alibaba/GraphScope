141  inline Result<std::pair<uint8_t, std::string_view>> parse_query_type(
142  const std::string& input) {
143  const char* str_data = input.data();
144  VLOG(10) << "parse query type for " << input << " size: " << input.size();
145  char input_tag = input.back();
146  VLOG(10) << "input tag: " << static_cast<int>(input_tag);
147  size_t len = input.size();
148  if (input_tag == static_cast<uint8_t>(InputFormat::kCppEncoder)) {
149  // For cpp encoder, the query id is the second last byte, others are all
150  // user-defined payload,
151  return std::make_pair((uint8_t) input[len - 2],
152  std::string_view(str_data, len - 2));
153  } else if (input_tag ==
154  static_cast<uint8_t>(InputFormat::kCypherProtoAdhoc)) {
155  // For cypher internal adhoc, the query id is the
156  // second last byte,which is fixed to 255, and other bytes are a string
157  // representing the path to generated dynamic lib.
158  return std::make_pair((uint8_t) input[len - 2],
159  std::string_view(str_data, len - 2));
160  } else if (input_tag == static_cast<uint8_t>(InputFormat::kCypherJson)) {
161  // For cypherJson there is no query-id provided. The query name is
162  // provided in the json string.
163  // We don't discard the last byte, since we need it to determine the input
164  // format when deserializing the input arguments in deserialize() function
165  std::string_view str_view(input.data(), len);
166  return parse_query_type_from_cypher_json(str_view);
167  } else if (input_tag ==
168  static_cast<uint8_t>(InputFormat::kCypherProtoProcedure)) {
169  // For cypher internal procedure, the query_name is
170  // provided in the protobuf message.
171  // Same as cypherJson, we don't discard the last byte.
172  std::string_view str_view(input.data(), len);
173  return parse_query_type_from_cypher_internal(str_view);
174  } else {
175  return Result<std::pair<uint8_t, std::string_view>>(
176  gs::Status(StatusCode::INVALID_ARGUMENT,
177  "Invalid input tag: " + std::to_string(input_tag)));
178  }
179  }
