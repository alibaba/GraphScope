854  threads.emplace_back([&]() {

856  while (true) {
857  size_t begin = std::min(v_i.fetch_add(chunk), vnum);
858  size_t end = std::min(begin + chunk, vnum);

864  ret += nbr.neighbor;
865  ++begin;
866  }
867  }
868  output.fetch_add(ret);
869  });
