28 class LDBCTimeStampParser : public arrow::TimestampParser {
29  public:
30  LDBCTimeStampParser() = default;
31 
32  ~LDBCTimeStampParser() override {}
33 
34  bool operator()(const char* s, size_t length, arrow::TimeUnit::type out_unit,
35  int64_t* out) const override {
36  using seconds_type = std::chrono::duration<arrow::TimestampType::c_type>;
37 
38  // We allow the following formats for all units:
39  // - "YYYY-MM-DD"
40  // - "YYYY-MM-DD[ T]hhZ?"
41  // - "YYYY-MM-DD[ T]hh:mmZ?"
42  // - "YYYY-MM-DD[ T]hh:mm:ssZ?"
43  //
44  // We allow the following formats for unit == MILLI, MICRO, or NANO:
45  // - "YYYY-MM-DD[ T]hh:mm:ss.s{1,3}Z?"
46  //
47  // We allow the following formats for unit == MICRO, or NANO:
48  // - "YYYY-MM-DD[ T]hh:mm:ss.s{4,6}Z?"
49  //
50  // We allow the following formats for unit == NANO:
51  // - "YYYY-MM-DD[ T]hh:mm:ss.s{7,9}Z?"
52  //
53  // UTC is always assumed, and the DataType's timezone is ignored.
54  //
55 
56  if (ARROW_PREDICT_FALSE(length < 10))
57  return false;
58 
59  seconds_type seconds_since_epoch;
60  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseYYYY_MM_DD(
61  s, &seconds_since_epoch))) {
62  return false;
63  }
64 
65  if (length == 10) {
66  *out =
67  arrow::util::CastSecondsToUnit(out_unit, seconds_since_epoch.count());
68  return true;
69  }
70 
71  if (ARROW_PREDICT_FALSE(s[10] != ' ') &&
72  ARROW_PREDICT_FALSE(s[10] != 'T')) {
73  return false;
74  }
75 
76  if (s[length - 1] == 'Z') {
77  --length;
78  }
79 
80  // if the back the s is '+0000', we should ignore it
81  auto time_zones = std::string_view(s + length - 5, 5);
82  if (time_zones == "+0000") {
83  length -= 5;
84  }
85 
86  seconds_type seconds_since_midnight;
87  switch (length) {
88  case 13: // YYYY-MM-DD[ T]hh
89  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseHH(
90  s + 11, &seconds_since_midnight))) {
91  return false;
92  }
93  break;
94  case 16: // YYYY-MM-DD[ T]hh:mm
95  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseHH_MM(
96  s + 11, &seconds_since_midnight))) {
97  return false;
98  }
99  break;
100  case 19: // YYYY-MM-DD[ T]hh:mm:ss
101  case 21: // YYYY-MM-DD[ T]hh:mm:ss.s
102  case 22: // YYYY-MM-DD[ T]hh:mm:ss.ss
103  case 23: // YYYY-MM-DD[ T]hh:mm:ss.sss
104  case 24: // YYYY-MM-DD[ T]hh:mm:ss.ssss
105  case 25: // YYYY-MM-DD[ T]hh:mm:ss.sssss
106  case 26: // YYYY-MM-DD[ T]hh:mm:ss.ssssss
107  case 27: // YYYY-MM-DD[ T]hh:mm:ss.sssssss
108  case 28: // YYYY-MM-DD[ T]hh:mm:ss.ssssssss
109  case 29: // YYYY-MM-DD[ T]hh:mm:ss.sssssssss
110  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseHH_MM_SS(
111  s + 11, &seconds_since_midnight))) {
112  return false;
113  }
114  break;
115  default:
116  LOG(ERROR) << "unsupported length: " << length;
117  return false;
118  }
119 
120  seconds_since_epoch += seconds_since_midnight;
121 
122  if (length <= 19) {
123  *out =
124  arrow::util::CastSecondsToUnit(out_unit, seconds_since_epoch.count());
125  return true;
126  }
127 
128  if (ARROW_PREDICT_FALSE(s[19] != '.')) {
129  return false;
130  }
131 
132  uint32_t subseconds = 0;
133  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseSubSeconds(
134  s + 20, length - 20, out_unit, &subseconds))) {
135  return false;
136  }
137 
138  *out =
139  arrow::util::CastSecondsToUnit(out_unit, seconds_since_epoch.count()) +
140  subseconds;
141  return true;
142  }
143 
144  const char* kind() const override { return "LDBC timestamp parser"; }
145 
146  const char* format() const override { return "EmptyFormat"; }
147 };

149 class LDBCLongDateParser : public arrow::TimestampParser {
150  public:
151  using seconds_type = std::chrono::duration<arrow::TimestampType::c_type>;
152  LDBCLongDateParser() = default;
153 
154  ~LDBCLongDateParser() override {}
155 
156  bool operator()(const char* s, size_t length, arrow::TimeUnit::type out_unit,
157  int64_t* out) const override {
158  uint64_t seconds;
159  // convert (s, s + length - 4) to seconds_since_epoch
160  if (ARROW_PREDICT_FALSE(
161  !arrow::internal::ParseUnsigned(s, length - 3, &seconds))) {
162  return false;
163  }
164 
165  uint32_t subseconds = 0;
166  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseSubSeconds(
167  s + length - 3, 3, out_unit, &subseconds))) {
168  return false;
169  }
170 
171  *out = arrow::util::CastSecondsToUnit(out_unit, seconds) + subseconds;
172  return true;
173  }
174 
175  const char* kind() const override { return "LDBC timestamp parser"; }
176 
177  const char* format() const override { return "LongDateFormat"; }
178 };
